1️⃣ Rôle de chaque dossier (important)
Avec ta structure :


modele/	Classes métier (représentent les tables)
dao/	Accès à la base de données (CRUD)

config/	Connexion DB, constantes
util/	Helpers (date, sécurité, validation…)

<?php

require_once __DIR__ . '/../config/Database.php';
require_once __DIR__ . '/../modele/Utilisateur.php';

class UtilisateurDAO
{
    private PDO $db;

    public function __construct()
    {
        $this->db = Database::getConnection();
    }

    public function findById(int $id): ?Utilisateur
    {
        $sql = "SELECT * FROM utilisateurs WHERE id_utilisateur = :id";
        $stmt = $this->db->prepare($sql);
        $stmt->execute(['id' => $id]);

        $row = $stmt->fetch(PDO::FETCH_ASSOC);

        if (!$row) {
            return null;
        }

        return new Utilisateur(
            $row['id_utilisateur'],
            $row['nom'],
            $row['prenom'],
            $row['email'],
            $row['mot_de_passe'],
            $row['role'],
            $row['date_creation']
        );
    }

    public function create(Utilisateur $utilisateur): bool
    {
        $sql = "INSERT INTO utilisateurs 
                (nom, prenom, email, mot_de_passe, role)
                VALUES (:nom, :prenom, :email, :mdp, :role)";

        $stmt = $this->db->prepare($sql);

        return $stmt->execute([
            'nom' => $utilisateur->getNom(),
            'prenom' => $utilisateur->getPrenom(),
            'email' => $utilisateur->getEmail(),
            'mdp' => $utilisateur->getMotDePasse(),
            'role' => $utilisateur->getRole()
        ]);
    }

    public function update(Utilisateur $utilisateur): bool
    {
        $sql = "UPDATE utilisateurs SET 
                    nom = :nom,
                    prenom = :prenom,
                    email = :email,
                    role = :role
                WHERE id_utilisateur = :id";

        $stmt = $this->db->prepare($sql);

        return $stmt->execute([
            'nom' => $utilisateur->getNom(),
            'prenom' => $utilisateur->getPrenom(),
            'email' => $utilisateur->getEmail(),
            'role' => $utilisateur->getRole(),
            'id' => $utilisateur->getIdUtilisateur()
        ]);
    }

    public function delete(int $id): bool
    {
        $sql = "DELETE FROM utilisateurs WHERE id_utilisateur = :id";
        $stmt = $this->db->prepare($sql);
        return $stmt->execute(['id' => $id]);
    }
}

🧩 ÉTAPE 3 — DAO (dao/)
👉 TOUT le SQL est ici
👉 Transforme les données DB ↔ objets

📄 dao/UtilisateurDAO.php
<?php

require_once __DIR__ . '/../config/Database.php';
require_once __DIR__ . '/../modele/Utilisateur.php';

class UtilisateurDAO
{
    private PDO $db;

    public function __construct()
    {
        $this->db = Database::getConnection();
    }

    public function findById(int $id): ?Utilisateur
    {
        $sql = "SELECT * FROM utilisateurs WHERE id_utilisateur = :id";
        $stmt = $this->db->prepare($sql);
        $stmt->execute(['id' => $id]);

        $row = $stmt->fetch(PDO::FETCH_ASSOC);

        if (!$row) return null;

        return new Utilisateur(
            $row['id_utilisateur'],
            $row['nom'],
            $row['prenom'],
            $row['email'],
            $row['role']
        );
    }

    public function create(Utilisateur $u): bool
    {
        $sql = "INSERT INTO utilisateurs (nom, prenom, email, role)
                VALUES (:nom, :prenom, :email, :role)";

        return $this->db->prepare($sql)->execute([
            'nom' => $u->getNom(),
            'prenom' => $u->getPrenom(),
            'email' => $u->getEmail(),
            'role' => $u->getRole()
        ]);
    }
}

<?php

require_once __DIR__ . '/../dao/UtilisateurDAO.php';

class UtilisateurService
{
    private UtilisateurDAO $dao;

    public function __construct()
    {
        $this->dao = new UtilisateurDAO();
    }

    public function creerUtilisateur(
        string $nom,
        string $prenom,
        string $email
    ): bool {
        if (empty($email)) {
            throw new Exception("Email obligatoire");
        }

        $utilisateur = new Utilisateur(
            null,
            $nom,
            $prenom,
            $email,
            'ELEVE'
        );

        return $this->dao->create($utilisateur);
    }

    public function getUtilisateur(int $id): ?Utilisateur
    {
        return $this->dao->findById($id);
    }
}

🧩 ÉTAPE 2 — DAO
📄 dao/InscriptionDAO.php

👉 SQL pur
👉 Aucune logique métier

<?php

require_once __DIR__ . '/../config/Database.php';
require_once __DIR__ . '/../modele/Inscription.php';

class InscriptionDAO
{
    private PDO $db;

    public function __construct()
    {
        $this->db = Database::getConnection();
    }

    public function existe(int $idUtilisateur, int $idClasse): bool
    {
        $sql = "SELECT COUNT(*) FROM inscriptions 
                WHERE id_utilisateur = :u AND id_classe = :c";

        $stmt = $this->db->prepare($sql);
        $stmt->execute([
            'u' => $idUtilisateur,
            'c' => $idClasse
        ]);

        return $stmt->fetchColumn() > 0;
    }

    public function create(Inscription $i): bool
    {
        $sql = "INSERT INTO inscriptions
                (id_utilisateur, id_classe, date_inscription, statut)
                VALUES (:u, :c, :d, :s)";

        return $this->db->prepare($sql)->execute([
            'u' => $i->getIdUtilisateur(),
            'c' => $i->getIdClasse(),
            'd' => date('Y-m-d'),
            's' => $i->getStatut()
        ]);
    }
}

🧩 ÉTAPE 3 — SERVICE (LOGIQUE MÉTIER)
📄 service/InscriptionService.php

👉 C’EST ICI QUE TOUT SE JOUE

<?php

require_once __DIR__ . '/../dao/InscriptionDAO.php';
require_once __DIR__ . '/../dao/UtilisateurDAO.php';
require_once __DIR__ . '/../dao/ClasseDAO.php';

class InscriptionService
{
    private InscriptionDAO $inscriptionDAO;
    private UtilisateurDAO $utilisateurDAO;
    private ClasseDAO $classeDAO;

    public function __construct()
    {
        $this->inscriptionDAO = new InscriptionDAO();
        $this->utilisateurDAO = new UtilisateurDAO();
        $this->classeDAO = new ClasseDAO();
    }

    public function inscrireEleve(int $idUtilisateur, int $idClasse): void
    {
        // 1️⃣ Vérifier que l'utilisateur existe
        if ($this->utilisateurDAO->findById($idUtilisateur) === null) {
            throw new Exception("Utilisateur inexistant");
        }

        // 2️⃣ Vérifier que la classe existe
        if ($this->classeDAO->findById($idClasse) === null) {
            throw new Exception("Classe inexistante");
        }

        // 3️⃣ Vérifier s'il est déjà inscrit
        if ($this->inscriptionDAO->existe($idUtilisateur, $idClasse)) {
            throw new Exception("Utilisateur déjà inscrit");
        }

        // 4️⃣ Créer l’inscription
        $inscription = new Inscription(
            null,
            $idUtilisateur,
            $idClasse,
            date('Y-m-d'),
            'ACTIVE'
        );

        $this->inscriptionDAO->create($inscription);
    }
}

<?php

require_once __DIR__ . '/../service/InscriptionService.php';

class InscriptionController
{
    private InscriptionService $service;

    public function __construct()
    {
        $this->service = new InscriptionService();
    }

    public function inscrire(): void
    {
        try {
            $this->service->inscrireEleve(
                $_POST['id_utilisateur'],
                $_POST['id_classe']
            );

            echo json_encode([
                'status' => 'success',
                'message' => 'Élève inscrit avec succès'
            ]);

        } catch (Exception $e) {
            http_response_code(400);
            echo json_encode([
                'status' => 'error',
                'message' => $e->getMessage()
            ]);
        }
    }
}
